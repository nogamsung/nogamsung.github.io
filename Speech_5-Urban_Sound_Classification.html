2. CNN Model

def cnn_model_01(input_dim):
    model = Sequential()

    model.add(Conv2D(filters=16, kernel_size=2, input_shape=input_dim, activation='relu'))
    modle.add(MaxPooling2D(pool_size=2))
    model.add(Conv2D(32, 2, padding='same', activation='relu'))
    model.add(MaxPooling2D(pool_size=2))
    model.add(Dropout(0.3))
    model.add(Conv2D(64, 2, activation='relu'))
    model.add(MaxPooling2D(pool_size=2))
    model.add(Conv2D(128, 2, activation='relu'))
    model.add(MaxPooling2D(pool_size=2))
    model.add(Dropout(0.3))
    model.add(GlobalAveragePooling2D())
    model.add(Flatten())
    model.add(Dense(1024, activation='relu'))
    model.add(Dense(512, activation='relu'))
    model.add(Dense(10, activation='softmax'))

    return model

model.compile

  • Environment settings for training method
    • loss: error rate function
    • metrics: performance assessment
    • optimizer: optmization method 
      model = cnn_model_01(input_dim)
model.compile(loss='categorical_crossentropy', metrics=['accuracy'], optimizer='adam')
